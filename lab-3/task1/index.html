    Упаковка FastAPI приложения, базы данных и парсера данных в Docker

    lab3/task1/Dockerfile
    FROM python:3.11

WORKDIR /app

COPY ../../lab1/requirements.txt .

RUN pip install --root-user-action=ignore -r requirements.txt

COPY ../../lab1 .

ENTRYPOINT ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8080"]
    lab3/task2_parser/Dockerfile
    FROM python:3.11

WORKDIR /app

COPY ../../lab1/requirements.txt .

RUN pip install --root-user-action=ignore -r requirements.txt

COPY ../../lab3/task2_parser .

ENTRYPOINT ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8081"]
    lab3/docker-compose.yaml
    version: '3.8'

services:
  task1_db:
    image: postgres:16.2
    restart: always
    environment:
      POSTGRES_PASSWORD: 12345
      POSTGRES_USER: postgres
      POSTGRES_DB: db
      PGPORT: 10000

  task1_server:
    build:
      context: ..
      dockerfile: lab3/task1/Dockerfile
    ports:
      - "8080:8080"
    env_file:
      - task1/.env
    depends_on:
      - task1_db

  task2_db:
    image: postgres:16.2
    restart: always
    environment:
      POSTGRES_PASSWORD: 12345
      POSTGRES_USER: postgres
      POSTGRES_DB: db
      PGPORT: 10001
    ports:
      - "10001:10001"

  task2_parser:
    build:
      context: ..
      dockerfile: lab3/task2_parser/Dockerfile
    ports:
      - "8081:8081"
    depends_on:
      - task2_db

  task2_proxy:
    build:
      context: ..
      dockerfile: lab3/task2_proxy/Dockerfile
    ports:
      - "8082:8082"
    depends_on:
      - task2_parser
      - task2_db
